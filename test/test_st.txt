
PROGRAM MAIN 
  VAR
    INPUT : BYTE ;
    OUTPUT : BYTE ;
  END_VAR

  INPUT := 1 ;
  OUTPUT := INPUT + 1 ;
  OUTPUT := FUNCTEST(INPUT, OUTPUT);
END_PROGRAM

FUNCTION FUNCTEST : INT
	VAR_INPUT
		A : INT;
		B : INT;
	END_VAR

  VAR
    C : INT;
  END_VAR

  C := A + B;
	FUNCTEST := C * 2;
END_FUNCTION

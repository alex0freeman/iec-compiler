FUNCTION FUNC : INT
	VAR_INPUT
		A, B : REAL := 1.56 ;
		C : INT ;
	END_VAR

	VAR
		D, E : INT := 5;
		F : INT ;
		G : BOOL := BOOL#false;
	END_VAR
	
	REPEAT
		C := C + 1;
		REPEAT
			D := D + 1;
			IF D > 4 THEN
				EXIT;
			END_IF;
		UNTIL D = 10
		END_REPEAT;
	UNTIL C = 100
	END_REPEAT;

	C := A  ;

END_FUNCTION

PROGRAM MAIN
	VAR_INPUT
		IN : INT ;
	END_VAR

	VAR_OUTPUT
		C : INT ;
	END_VAR
	

	C := IN + 2;

END_PROGRAM



PROGRAM MAIN1
	VAR_INPUT
		IN : INT ;
	END_VAR

	VAR_OUTPUT
		C : INT ;
	END_VAR
	

	C := IN + 2;

END_PROGRAM


CONFIGURATION MAIN_CONFIG

	RESOURCE RES1 ON CPU1
		VAR_GLOBAL
			Z : INT;
		END_VAR

		TASK fast(INTERVAL := T#10ms, PRIORITY := 4);
		TASK slow(SINGLE := Z, PRIORITY := 2);

		PROGRAM P1 WITH fast: MAIN(IN := 3);
		PROGRAM P2 WITH slow: MAIN1(C => Z, IN := 2);

	END_RESOURCE
	
END_CONFIGURATION
